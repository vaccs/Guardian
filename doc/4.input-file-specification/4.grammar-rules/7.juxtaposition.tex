
\subsection{Juxtaposition}
{
	Juxtaposition can be used between two grammar-rule patterns to
	"concatenate" their patterns.
	
	\begin{itemize}
	{
		\item[] \lstinline[language=MAIA, columns=fixed]@"a" ("b" | "c") "d"@
		
			The token ``a", either the token ``b" or the token ``c",
			then the token ``d".
		
		\item[] \lstinline[language=MAIA, columns=fixed]@"a"* "b"+@
		
			Zero or more ``a" tokens, then then one or more ``b" tokens.
	}
	\end{itemize}
}
